```python
from __future__ import annotations

from sqlalchemy import text
from sqlalchemy.ext.asyncio import AsyncSession


async def get_audit_trail(db: AsyncSession, document_id: str) -> dict:
    """Return ordered audit trail for a document's processing pipeline."""
    query = text(
        """
        SELECT id, document_id, field_id, step_name, step_order,
               input_summary, output_summary, parameters,
               duration_ms, created_at
        FROM audit_log
        WHERE document_id = :document_id
        ORDER BY step_order ASC, created_at ASC
        """
    )
    result = await db.execute(query, {"document_id": document_id})
    rows = result.mappings().all()

    steps = [
        {
            "id": str(row["id"]),
            "document_id": str(row["document_id"]),
            "field_id": str(row["field_id"]) if row["field_id"] else None,
            "step_name": row["step_name"],
            "step_order": row["step_order"],
            "input_summary": row["input_summary"],
            "output_summary": row["output_summary"],
            "parameters": row["parameters"],
            "duration_ms": row["duration_ms"],
            "created_at": row["created_at"]
            if isinstance(row["created_at"], str)
            else row["created_at"].isoformat(),
        }
        for row in rows
    ]

    total_duration_ms = sum(s["duration_ms"] for s in steps)

    return {
        "document_id": str(document_id),
        "steps": steps,
        "total_duration_ms": total_duration_ms,
        "step_count": len(steps),
    }


async def get_extraction(db: AsyncSession, document_id: str) -> dict:
    """Return extraction results with all extracted fields."""
    result_query = text(
        """
        SELECT id, document_id, mode, template_type, raw_response,
               processing_time_ms, created_at
        FROM extraction_results
        WHERE document_id = :document_id
        ORDER BY created_at DESC
        LIMIT 1
        """
    )
    result = await db.execute(result_query, {"document_id": document_id})
    extraction = result.mappings().first()

    if extraction is None:
        return None

    fields_query = text(
        """
        SELECT id, field_type, field_key, field_value, page_number,
               bounding_box, confidence, vlm_confidence, cv_quality_score,
               is_required, is_missing
        FROM extracted_fields
        WHERE extraction_id = :extraction_id
        ORDER BY page_number ASC, field_key ASC
        """
    )
    fields_result = await db.execute(
        fields_query, {"extraction_id": str(extraction["id"])}
    )
    fields = fields_result.mappings().all()

    return {
        "id": str(extraction["id"]),
        "document_id": str(extraction["document_id"]),
        "mode": extraction["mode"],
        "template_type": extraction["template_type"],
        "processing_time_ms": extraction["processing_time_ms"],
        "created_at": extraction["created_at"]
        if isinstance(extraction["created_at"], str)
        else extraction["created_at"].isoformat(),
        "fields": [
            {
                "id": str(f["id"]),
                "field_type": f["field_type"],
                "field_key": f["field_key"],
                "field_value": f["field_value"],
                "page_number": f["page_number"],
                "bounding_box": f["bounding_box"],
                "confidence": f["confidence"],
                "vlm_confidence": f["vlm_confidence"],
                "cv_quality_score": f["cv_quality_score"],
                "is_required": f["is_required"],
                "is_missing": f["is_missing"],
            }
            for f in fields
        ],
    }
```

```python
router = APIRouter(prefix="/api/documents", tags=["extractions"])


@router.get("/{document_id}/extraction")

    document_id: str,
    user: dict = Depends(get_current_user),
    db: AsyncSession = Depends(get_db),

        return {"error": "Extraction not found", "document_id": document_id}

    document_id: str,
    user: dict = Depends(get_current_user),
    db: AsyncSession = Depends(get_db),
):
```

```python
# Add to backend/app/services/extraction_service.py

def _confidence_color(confidence: float) -> str:
    """Map confidence score to color: green (>=0.8), yellow (0.5-0.8), red (<0.5)."""
    if confidence >= 0.8:
        return "green"
    elif confidence >= 0.5:
        return "yellow"
    return "red"


def _generate_tooltip(
    field: dict, quality_regions: list[dict] | None
) -> str | None:
    """Generate tooltip for low/medium confidence fields explaining the reason."""
    confidence = field["confidence"]
    if confidence >= 0.8:
        return None

    reasons = []
    cv_score = field.get("cv_quality_score", 1.0)

    if cv_score < 0.4:
        reasons.append("Low image quality in this region")
    if cv_score < 0.5:
        reasons.append("Possible blur or noise detected")

    vlm_conf = field.get("vlm_confidence", 1.0)
    if vlm_conf < 0.6:
        reasons.append("Model uncertainty on extracted value")

    if quality_regions:
        bbox = field.get("bounding_box", {})
        for region in quality_regions:
            if region.get("blur_score", 1.0) < 0.4:
                reasons.append("Blur detected in source region")
                break
            if region.get("noise_score", 1.0) < 0.4:
                reasons.append("Noise detected in source region")
                break
            if region.get("contrast_score", 1.0) < 0.4:
                reasons.append("Low contrast in source region")
                break

    if not reasons:
        reasons.append("Reduced confidence — verify manually")

    return "; ".join(dict.fromkeys(reasons))


async def get_overlay_data(db: AsyncSession, document_id: str) -> dict:
    """Return confidence overlay data for document viewer heatmap."""
    fields_query = text(
        """
        SELECT ef.id, ef.field_key, ef.field_value, ef.page_number,
               ef.bounding_box, ef.confidence, ef.vlm_confidence,
               ef.cv_quality_score
        FROM extracted_fields ef
        JOIN extraction_results er ON ef.extraction_id = er.id
        WHERE er.document_id = :document_id
        ORDER BY ef.page_number ASC
        """
    )
    fields_result = await db.execute(fields_query, {"document_id": document_id})
    fields = fields_result.mappings().all()

    quality_query = text(
        """
        SELECT output_summary
        FROM audit_log
        WHERE document_id = :document_id AND step_name = 'quality_assessment'
        ORDER BY created_at DESC
        LIMIT 1
        """
    )
    quality_result = await db.execute(quality_query, {"document_id": document_id})
    quality_row = quality_result.mappings().first()
    quality_regions = (
        quality_row["output_summary"].get("regions", [])
        if quality_row and quality_row.get("output_summary")
        else None
    )

    field_overlays = []
    for field in fields:
        field_dict = dict(field)
        color = _confidence_color(field_dict["confidence"])
        tooltip = _generate_tooltip(field_dict, quality_regions)

        field_overlays.append(
            {
                "field_id": str(field_dict["id"]),
                "field_key": field_dict["field_key"],
                "page_number": field_dict["page_number"],
                "bounding_box": field_dict["bounding_box"],
                "confidence": field_dict["confidence"],
                "color": color,
                "tooltip": tooltip,
            }
        )

    return {
        "document_id": str(document_id),
        "field_overlays": field_overlays,
        "summary": {
            "total_fields": len(field_overlays),
            "high_confidence": len([f for f in field_overlays if f["color"] == "green"]),
            "medium_confidence": len([f for f in field_overlays if f["color"] == "yellow"]),
            "low_confidence": len([f for f in field_overlays if f["color"] == "red"]),
        },
    }
```

```python
# Add to backend/app/services/extraction_service.py

async def get_comparison(db: AsyncSession, document_id: str) -> dict | None:
    """Compare enhanced extraction results against raw VLM baseline."""
    enhanced_query = text(
        """
        SELECT ef.field_key, ef.field_value, ef.confidence, ef.bounding_box
        FROM extracted_fields ef
        JOIN extraction_results er ON ef.extraction_id = er.id
        WHERE er.document_id = :document_id
        ORDER BY ef.field_key ASC
        """
    )
    enhanced_result = await db.execute(enhanced_query, {"document_id": document_id})
    enhanced_fields = enhanced_result.mappings().all()

    baseline_query = text(
        """
        SELECT raw_fields
        FROM raw_baselines
        WHERE document_id = :document_id
        ORDER BY created_at DESC
        LIMIT 1
        """
    )
    baseline_result = await db.execute(baseline_query, {"document_id": document_id})
    baseline_row = baseline_result.mappings().first()

    if baseline_row is None:
        return {"error": "No raw baseline available for comparison", "document_id": str(document_id)}

    raw_fields = baseline_row["raw_fields"]
    raw_by_key = {f["field_key"]: f for f in raw_fields}

    diffs = []
    enhanced_keys = set()

    for field in enhanced_fields:
        field_dict = dict(field)
        key = field_dict["field_key"]
        enhanced_keys.add(key)

        if key in raw_by_key:
            raw = raw_by_key[key]
            if field_dict["field_value"] != raw["field_value"]:
                status = "corrected"
            else:
                status = "unchanged"
            diffs.append(
                {
                    "field_key": key,
                    "status": status,
                    "enhanced_value": field_dict["field_value"],
                    "raw_value": raw["field_value"],
                    "enhanced_confidence": field_dict["confidence"],
                    "raw_confidence": raw.get("confidence"),
                    "bounding_box": field_dict["bounding_box"],
                }
            )
        else:
            diffs.append(
                {
                    "field_key": key,
                    "status": "added",
                    "enhanced_value": field_dict["field_value"],
                    "raw_value": None,
                    "enhanced_confidence": field_dict["confidence"],
                    "raw_confidence": None,
                    "bounding_box": field_dict["bounding_box"],
                }
            )

    corrected_count = len([d for d in diffs if d["status"] == "corrected"])
    added_count = len([d for d in diffs if d["status"] == "added"])
    unchanged_count = len([d for d in diffs if d["status"] == "unchanged"])

    return {
        "document_id": str(document_id),
        "diffs": diffs,
        "summary": {
            "corrected": corrected_count,
            "added": added_count,
            "unchanged": unchanged_count,
            "total": len(diffs),
        },
    }
```
